CREATE TABLE `users` (   `id` int(10) UNSIGNED PRIMARY KEY NOT NULL AUTO_INCREMENT,

                           `location` varchar(50) DEFAULT NULL ) ENGINE=InnoDB DEFAULT CHARSET=latin1;
